#include <Arduino.h>
#include <ESP8266WiFi.h>
#include <Firebase_ESP_Client.h>

// นำเข้าโมดูลช่วยเหลือของ Firebase สำหรับจัดการ Token และ RTDB
#include "addons/TokenHelper.h"
#include "addons/RTDBHelper.h"

// ==========================================
// 1. ตั้งค่า Wi-Fi (ระบุข้อมูลของคุณตรงนี้)
// ==========================================
#define WIFI_SSID "Porn"
#define WIFI_PASSWORD "1234567899"

// ==========================================
// 2. ข้อมูล Firebase จากโปรเจกต์ของคุณ (ใส่ให้เรียบร้อยแล้ว)
// ==========================================
#define API_KEY "AIzaSyCTGTfRQAqBdUzxN13YME0Zig_ifCZZuFs"
#define DATABASE_URL "https://yout2026-default-rtdb.asia-southeast1.firebasedatabase.app"

// ประกาศออบเจ็กต์สำหรับ Firebase
FirebaseData fbdo;
FirebaseAuth auth;
FirebaseConfig config;

void setup() {
  // เปิด Serial Monitor ที่ Baud Rate 115200
  Serial.begin(115200);
  delay(1000);
  
  Serial.println("\n\n========================================");
  Serial.println("  IoT Smart Home - SYSTEM DIAGNOSTIC   ");
  Serial.println("========================================");

  // ----------------------------------------
  // ข้อที่ 1: ตรวจสอบระบบ Wi-Fi
  // ----------------------------------------
  Serial.print("[Wi-Fi] กำลังเชื่อมต่อกับ: ");
  Serial.println(WIFI_SSID);
  
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  
  Serial.println("\n[Wi-Fi] เชื่อมต่อสำเร็จ!");
  Serial.print("[Wi-Fi] IP Address ที่ได้รับ: ");
  Serial.println(WiFi.localIP());
  Serial.println("----------------------------------------");

  // ----------------------------------------
  // ระบบซิงค์เวลา NTP (จำเป็นมากสำหรับ SSL ของ Firebase)
  // ----------------------------------------
  Serial.print("[NTP] กำลังซิงค์เวลาสากลผ่านอินเทอร์เน็ต");
  configTime(7 * 3600, 0, "pool.ntp.org", "time.nist.gov"); // เวลาประเทศไทย UTC+7
  while (time(nullptr) < 100000) {
    Serial.print(".");
    delay(500);
  }
  Serial.println("\n[NTP] ซิงค์เวลาสำเร็จ!");
  Serial.println("----------------------------------------");

  // ----------------------------------------
  // ข้อที่ 2: เริ่มต้นระบบ Firebase
  // ----------------------------------------
  Serial.println("[Firebase] กำลังตั้งค่าการเชื่อมต่อ...");
  config.api_key = API_KEY;
  config.database_url = DATABASE_URL;

  // เชื่อมต่อแบบไม่ระบุตัวตน (Anonymous Sign-In)
  if (Firebase.signUp(&config, &auth, "", "")) {
    Serial.println("[Firebase] ยืนยันตัวตน (Authentication) สำเร็จ!");
  } else {
    Serial.printf("[Firebase] ยืนยันตัวตนล้มเหลว: %s\n", config.signer.signupError.message.c_str());
  }

  Firebase.begin(&config, &auth);
  Firebase.reconnectWiFi(true);
  Serial.println("========================================\n");
}

void loop() {
  // ตรวจสอบว่าบอร์ดพร้อม และ Firebase เชื่อมต่อได้หรือไม่
  if (Firebase.ready()) {
    
    // กำหนดพิกัดที่ต้องการตรวจสอบ (ในที่นี้เริ่มจากพิกัดราก "/" เพื่อดูโครงสร้างทั้งหมด)
    String targetPath = "/"; 
    
    Serial.print("[Firebase] กำลังทดสอบอ่านข้อมูลจากพิกัด: ");
    Serial.println(targetPath);

    // ทดลองดึงข้อมูลในรูปแบบ JSON มาแสดงผล
    if (Firebase.RTDB.getJSON(&fbdo, targetPath)) {
      Serial.println("[Firebase] ดึงค่าสำเร็จ! ข้อมูลในพิกัดนี้คือ:");
      // แสดงโครงสร้างข้อมูลทั้งหมดที่ดึงมาได้
      Serial.println(fbdo.jsonString()); 
    } else {
      // หากดึงไม่สำเร็จ จะพิมพ์แจ้งสาเหตุความล้มเหลวออกมา
      Serial.print("[Firebase] ดึงข้อมูลล้มเหลว! สาเหตุ: ");
      Serial.println(fbdo.errorReason());
    }

    Serial.println("----------------------------------------");
  } else {
    Serial.println("[System] บอร์ดยังไม่พร้อม หรืออยู่ระหว่างการเชื่อมต่อ Firebase...");
  }

  // หน่วงเวลาตรวจสอบทุกๆ 5 วินาที เพื่อไม่ให้ Serial Monitor ทำงานเร็วเกินไป
  delay(5000);
}
