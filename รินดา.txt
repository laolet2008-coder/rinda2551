    <script src="https://www.gstatic.com/firebasejs/8.10.1/firebase-app.js"></script>
    <script src="https://www.gstatic.com/firebasejs/8.10.1/firebase-database.js"></script>

    <script>
        // 1. นำการตั้งค่า Firebase ของคุณมาใส่ตรงนี้
        // (หาได้จาก Firebase Console -> Project Settings -> General -> Your apps)
        const firebaseConfig = {
            apiKey: "YOUR_API_KEY",
            authDomain: "YOUR_PROJECT_ID.firebaseapp.com",
            databaseURL: "https://YOUR_PROJECT_ID-default-rtdb.firebaseio.com",
            projectId: "YOUR_PROJECT_ID",
            storageBucket: "YOUR_PROJECT_ID.appspot.com",
            messagingSenderId: "YOUR_SENDER_ID",
            appId: "YOUR_APP_ID"
        };

        // 2. เริ่มต้นใช้งาน Firebase
        firebase.initializeApp(firebaseConfig);
        const database = firebase.database();

        // 3. ฟังก์ชันสำหรับอัปเดตสถานะขึ้น Firebase เมื่อผู้ใช้กดสวิตช์
        function toggleDevice(pin) {
            const isChecked = document.getElementById(`switch-${pin}`).checked;
            
            // ส่งค่าไปยัง Firebase ภายใต้โหนดชื่อ 'SmartHome'
            database.ref('SmartHome/' + pin).set(isChecked)
            .then(() => {
                console.log(`${pin} ถูกตั้งค่าเป็น ${isChecked}`);
            })
            .catch((error) => {
                console.error("เกิดข้อผิดพลาด: ", error);
            });
        }

        // 4. ฟังก์ชันสำหรับรับสถานะจาก Firebase แบบ Real-time มาแสดงบนหน้าเว็บ
        // (กรณีมีการเปิด/ปิดจากที่อื่น เช่น จากบอร์ด NodeMCU หรือเว็บเครื่องอื่น)
        function listenToDevice(pin) {
            database.ref('SmartHome/' + pin).on('value', (snapshot) => {
                const status = snapshot.val();
                
                // อัปเดตสถานะสวิตช์บนหน้าเว็บ
                if (status !== null) {
                    document.getElementById(`switch-${pin}`).checked = status;
                    
                    // เปลี่ยนสีไอคอนให้สว่างขึ้นเมื่อเปิด
                    const icon = document.getElementById(`icon-${pin}`);
                    if (status) {
                        icon.style.color = "#00f3ff";
                        icon.style.textShadow = "0 0 10px #00f3ff";
                    } else {
                        icon.style.color = "#9ca3af";
                        icon.style.textShadow = "none";
                    }
                }
            });
        }

        // เริ่มต้นดักจับข้อมูลจาก Firebase สำหรับทั้ง 4 ช่อง
        listenToDevice('D1');
        listenToDevice('D2');
        listenToDevice('D5');
        listenToDevice('D6');
    </script>
