    <script>
        // 1. วาง Firebase Configuration ของคุณที่นี่
        const firebaseConfig = {
            apiKey: "YOUR_API_KEY",
            authDomain: "YOUR_PROJECT_ID.firebaseapp.com",
            databaseURL: "https://YOUR_PROJECT_ID-default-rtdb.firebaseio.com",
            projectId: "YOUR_PROJECT_ID",
            storageBucket: "YOUR_PROJECT_ID.appspot.com",
            messagingSenderId: "YOUR_SENDER_ID",
            appId: "YOUR_APP_ID"
        };

        // เริ่มต้นระบบ Firebase
        firebase.initializeApp(firebaseConfig);
        const database = firebase.database();

        // 2. ฟังก์ชันส่งค่าสถานะไปยัง Firebase (ส่งเป็น 1 หรือ 0 เพื่อไม่ให้บอร์ดไมโครคอนโทรลเลอร์สับสน)
        function toggleDevice(pin) {
            const isChecked = document.getElementById(`switch-${pin}`).checked;
            const valueToSend = isChecked ? 1 : 0; // ส่ง 1=เปิด, 0=ปิด เพื่อให้ตรงกับสถานะบอร์ดไอโอที
            
            database.ref('SmartHome/' + pin).set(valueToSend)
            .then(() => {
                console.log(`ส่งข้อมูลสำเร็จ: ${pin} -> ${valueToSend}`);
            })
            .catch((error) => {
                console.error("Firebase เขียนข้อมูลล้มเหลว: ", error);
            });
        }

        // 3. ฟังก์ชันดักจับข้อมูลแบบ Real-time และแปลงค่าข้อมูลให้ปลอดภัย (ป้องกันบั๊กสวิตช์ค้าง)
        function listenToDevice(pin) {
            database.ref('SmartHome/' + pin).on('value', (snapshot) => {
                const status = snapshot.val();
                
                if (status !== null) {
                    // ตรวจสอบเงื่อนไขอย่างละเอียด ป้องกันการตีความข้อมูล String หรือ Integer ผิดพลาด
                    const isOn = (status === true || status === 1 || status === "1" || status === "true");
                    
                    // ปรับสถานะปุ่ม Toggle บนหน้าจอ
                    document.getElementById(`switch-${pin}`).checked = isOn;
                    
                    // ปรับเปลี่ยนสีและแสงเอฟเฟกต์ตามสถานะจริง
                    const icon = document.getElementById(`icon-${pin}`);
                    const row = document.getElementById(`row-${pin}`);
                    
                    if (isOn) {
                        icon.style.color = "#00f3ff";
                        icon.style.textShadow = "0 0 10px #00f3ff";
                        row.classList.add('active');
                    } else {
                        icon.style.color = "#8b949e";
                        icon.style.textShadow = "none";
                        row.classList.remove('active');
                    }
                }
            });
        }

        // เริ่มต้นการซิงค์ข้อมูลแบบ Real-time ทั้ง 4 ช่องสัญญาณ
        listenToDevice('D1');
        listenToDevice('D2');
        listenToDevice('D5');
        listenToDevice('D6');
    </script>
